3   import com.matkansuunnittelija.travelplanobjects.DayEvent;

7   /**
8    * Luokka joka vastaa HTML-sivun generaroinnista matkasuunnitelman pohjalta.
9    *
10   * @author Samuli
11   */
12  public class HTMLTravelPlanGenerator {
13  
14      static private String generateHTMLForDay(DayPlan plan) {
15          StringBuilder builder = new StringBuilder();
16          builder.append(HTMLBuilder.generateHeader3(plan.getName())).append("\n");
17          HTMLTableBuilder tableBuilder = new HTMLTableBuilder("Aika", "Nimi", "Kuvaus");
18          for (DayEvent event : plan.getDayEvents()) {
19              tableBuilder.addRowToTable(event.getTime(), event.getName(), event.getDescription());
20          }
21          builder.append(tableBuilder.generateTable()).append("\n");
22          return builder.toString();
23      }
24  
25      static private String setTableStyle() {
26          CSSStyleBuilder builder = new CSSStyleBuilder();
27          builder.addNewValueForStyle("table, th, td", "border", "1px solid black");
28          builder.addNewValueForStyle("table, th, td", "border-collapse", "collapse");
29          builder.addNewValueForStyle("th, td", "padding", "5px");
30          builder.addNewValueForStyle("th, td", "text-align", "left");
31          return builder.generateCSS();
32      }
33  
34      /**
35       * Generoi HTML-sivun matkasuunnitelman pohjalta.
36       * @param plan Matkasuunnitelma josta sivu generoidaan
37       * @return matkasuunnitelma HTML-sivuna
38       */
39      static public String generateHTMLPage(TravelPlan plan) {
40          StringBuilder webPageGenerator = new StringBuilder();
41          webPageGenerator.append(HTMLBuilder.generateHeader1(plan.getName() + " " + plan.getStartDateAsString() + " - " + plan.getEndDateAsString())).append("\n");
42          webPageGenerator.append(HTMLBuilder.generateLineBreak());
43          for (DayPlan planForDay : plan.getDayPlansAsList()) {
44              webPageGenerator.append(generateHTMLForDay(planForDay));
45          }
46          return HTMLBuilder.generateDocument(setTableStyle(), webPageGenerator.toString());
47      }
48  
49  }
