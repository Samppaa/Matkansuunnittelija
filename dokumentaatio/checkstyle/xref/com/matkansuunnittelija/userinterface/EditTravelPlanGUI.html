23      private void initListView() {
24          listViewModel = (DefaultListModel) jList1.getModel();
25          listViewModel.removeAllElements();
26          for (DayPlan d : plan.getDayPlansAsList()) {
27              listViewModel.addElement(d.getName());
28          }
29      }
